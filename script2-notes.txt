file-finder.sh

#!/bin/bash

fileName=
directoryName=
fileSizeGreaterThan=
fileSizeLessThan=
fileContent=

echo "Menu"
echo "1. Name (with extension)"
echo "2. Directory"
echo "3. Greater than (file size, bytes)"
echo "4. Less than (file size, bytes)"
echo "5. Content"
echo "6. Run search"
echo "7. Quit"

read userInput

while [ $userInput != "7" ]; do

case $userInput in

"1"|"1.")
echo "File name: "
read fileName
;;

"2"|"2.")
echo "Directory name (without slash at the end): "
read directoryName
;;

"3"|"3.")
echo "File size greater than [bytes]: "
read fileSizeGreaterThan
;;

"4"|"4.")
echo "File size less than [bytes]: "
read fileSizeLessThan
;;

"5"|"5.")
echo "File content: "
read fileContent
;;

"6"|"6.")
echo "Running search..."
find -path "${directoryName}/${fileName}"
;;

*)
echo "Wrong input"
;;
esac

echo "Menu"
echo "1. Name (with extension): ${fileName}"
echo "2. Directory: ${directoryName}"
echo "3. Greater than (file size, bytes): ${fileSizeGreaterThan}"
echo "4. Less than (file size, bytes): ${fileSizeLessThan}"
echo "5. Content: ${fileContent}"
echo "6. Run search"
echo "7. Quit"

read userInput

done
